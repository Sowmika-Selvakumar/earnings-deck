Code Sample: NPV in Python

from decimal import Decimal

def npv(rate, cashflows):
    rate = Decimal(str(rate))
    return sum(Decimal(str(cf)) / (1 + rate) ** t for t, cf in enumerate(cashflows, start=1))

print(npv(0.1, [100, 120, 150]))
